import java.awt.*;


class LoadImageApp extends Frame {
	Image i = getToolkit().getImage("c:/javalogo.gif");
		
	public void paint( Graphics g ) {
		g.drawImage(i, 0,0,this);
	}
	
	public static void main (String a[]) {
		LoadImageApp e = new LoadImageApp();
		setSize(200, 200);
		e.setVisible(true);
	}
}
